SYNOPSIS
subroutine glstsq(ideg,x,y,n0,d) 

   integer             :: ideg
   real                   x(*)
   real                   y(*)
   integer             :: n0
   doubleprecision        d(*)

DESCRIPTION
least squares fit to a polynomial expression

OPTIONS
x X values to fit
y Y values to fit
ideg is desired degree of least square fit and test
n0 is number of points in curve arrays (x, y)
d is returned coefficient array
